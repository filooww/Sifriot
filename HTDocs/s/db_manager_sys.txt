CREATE TABLE algorithms
(
  id_algorithm smallint unsigned NOT NULL,
  alg_offset tinyint unsigned NOT NULL DEFAULT '0',
  del_from_source tinyint unsigned NOT NULL DEFAULT '0',
  beg_limit_set varchar(255) CHARACTER SET utf8mb3 COLLATE utf8mb3_bin NOT NULL,
  beg_number tinyint unsigned NOT NULL DEFAULT '0',
  beg_inc tinyint unsigned NOT NULL DEFAULT '0',
  beg_scr tinyint unsigned NOT NULL DEFAULT '0',
  inner_limit_set varchar(255) CHARACTER SET utf8mb3 COLLATE utf8mb3_bin NOT NULL,
  end_limit_set varchar(255) CHARACTER SET utf8mb3 COLLATE utf8mb3_bin NOT NULL,
  end_number tinyint unsigned NOT NULL DEFAULT '0',
  end_inc tinyint unsigned NOT NULL DEFAULT '0',
  end_scr tinyint unsigned NOT NULL DEFAULT '0',
  del_symbols varchar(255) CHARACTER SET utf8mb3 COLLATE utf8mb3_bin NOT NULL,
  ins_symbols varchar(255) CHARACTER SET utf8mb3 COLLATE utf8mb3_bin NOT NULL,
  field_only tinyint unsigned NOT NULL DEFAULT '0',
  reg_expression varchar(255) CHARACTER SET utf8mb3 COLLATE utf8mb3_bin NOT NULL,
  reg_scr tinyint unsigned NOT NULL DEFAULT '0',
  alg_remarks varchar(255) CHARACTER SET utf8mb3 COLLATE utf8mb3_bin NOT NULL DEFAULT 'New algorithm'
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb3;
ALTER TABLE algorithms
  ADD PRIMARY KEY (id_algorithm),
  ADD UNIQUE KEY id_algorithm (id_algorithm);
ALTER TABLE algorithms
  MODIFY id_algorithm smallint unsigned NOT NULL AUTO_INCREMENT, AUTO_INCREMENT=1;
